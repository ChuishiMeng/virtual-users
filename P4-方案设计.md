#### 2.3.1 ResidualPersona (残差表示)

```python
@dataclass
class ResidualPersona:
    """
    Persona 残差表示：个体 = 群体基线 + 个体残差
    """
    id: str
    population_baseline: Dict[str, Distribution]  # 群体统计分布
    individual_residuals: Dict[str, float]        # 个体残差
    residual_variance: Dict[str, float]           # 残差方差
    
    def sample_attributes(self) -> Dict[str, Any]:
        """采样个体属性"""
        attributes = {}
        for attr, dist in self.population_baseline.items():
            base_value = dist.sample()
            if attr in self.individual_residuals:
                residual = self.individual_residuals[attr]
                variance = self.residual_variance.get(attr, 1.0)
                adjusted_value = base_value + residual * variance
                attributes[attr] = adjusted_value
            else:
                attributes[attr] = base_value
        return attributes
    
    @classmethod
    def from_data(cls, population_data: pd.DataFrame, 
                  individual_data: pd.DataFrame) -> 'ResidualPersona':
        """从数据构建残差 Persona"""
        # 拟合群体分布
        pop_baseline = {}
        for col in population_data.columns:
            pop_baseline[col] = fit_distribution(population_data[col])
        
        # 计算个体残差
        residuals = {}
        variances = {}
        for col in population_data.columns:
            pop_mean = population_data[col].mean()
            ind_value = individual_data[col].iloc[0]
            residuals[col] = ind_value - pop_mean
            variances[col] = population_data[col].std()
        
        return cls(pop_baseline, residuals, variances)
```

#### 2.3.2 ConsistAgent (一致性约束解码)

```python
class ConsistAgent:
    """
    一致性约束虚拟用户代理
    """
    def __init__(self, 
                 constraint_strength: float = 0.8,
                 memory_window: int = 10,
                 constraints: List[ConsistencyConstraint] = None):
        self.constraint_strength = constraint_strength  # α参数
        self.memory_window = memory_window
        self.memory = ConsistencyMemory(max_size=memory_window)
        self.constraints = constraints or []
        self.retriever = ConstraintRetriever(constraints)
        self.decoder = WeightedDecoder(constraint_strength)
    
    def generate_response(self, 
                         question: SurveyQuestion,
                         persona: ResidualPersona,
                         context: Optional[str] = None) -> SurveyResponse:
        """生成一致性约束的响应"""
        # 1. 采样 Persona 属性
        attributes = persona.sample_attributes()
        
        # 2. 检索相关约束
        relevant_constraints = self.retriever.retrieve(
            question.id, 
            self.memory.responses
        )
        
        # 3. 约束加权解码
        weighted_options = self.decoder.apply_constraints(
            question.options,
            relevant_constraints,
            self.memory.responses,
            attributes
        )
        
        # 4. 加权采样
        response = self.decoder.weighted_sample(weighted_options)
        
        # 5. 更新记忆
        self.memory.update(question.id, response)
        
        return SurveyResponse(
            question_id=question.id,
            response=response,
            confidence=max(weighted_options.values()),
            metadata={
                'constraints_applied': len(relevant_constraints),
                'persona_attributes': attributes
            }
        )
```

#### 2.3.3 ACSMetric (一致性评估)

```python
class ACSMetric:
    """
    Attitude Consistency Score - 态度一致性分数
    """
    def evaluate(self, 
                responses: List[SurveyResponse],
                constraints: List[ConsistencyConstraint]) -> ACSResult:
        """
        评估一致性
        
        Returns:
            ACSResult:
                - overall_score: 总体 ACS
                - domain_scores: 各领域 ACS (政治/社会/人口)
                - constraint_scores: 各约束级得分
                - bootstrap_ci: Bootstrap 置信区间
        """
        # 计算约束级得分
        constraint_scores = {}
        for constraint in constraints:
            score = self._check_constraint(responses, constraint)
            constraint_scores[constraint.id] = score
        
        # 计算领域级得分
        domain_scores = self._aggregate_by_domain(constraint_scores)
        
        # 计算总体得分
        overall_score = np.mean(list(constraint_scores.values()))
        
        # Bootstrap 置信区间
        bootstrap_ci = self._bootstrap_ci(responses, constraints)
        
        return ACSResult(
            overall_score=overall_score,
            domain_scores=domain_scores,
            constraint_scores=constraint_scores,
            bootstrap_ci=bootstrap_ci
        )
```

### 2.4 关键算法

#### 算法 1: 约束加权解码

```python
def constrained_decode(options: List[str],
                      constraints: List[ConsistencyConstraint],
                      memory_responses: Dict[str, Any],
                      alpha: float = 0.8) -> Dict[str, float]:
    """
    约束加权解码算法
    
    Args:
        options: 候选选项列表
        constraints: 相关约束列表
        memory_responses: 历史响应记忆
        alpha: 约束强度 (0=无约束，1=完全约束)
    
    Returns:
        选项权重字典 {option: weight}
    """
    # 初始化均匀权重
    weights = {opt: 1.0 for opt in options}
    
    for constraint in constraints:
        # 获取相关问题及其响应
        other_q = constraint.get_other_question(question_id)
        if other_q not in memory_responses:
            continue
        
        other_response = memory_responses[other_q]
        
        for option in options:
            # 检查一致性
            is_consistent = check_consistency(
                option, other_response, constraint
            )
            
            if constraint.type == "logical":
                # 硬约束：不一致则权重归零
                if not is_consistent:
                    weights[option] = 0.0
            else:
                # 软约束：按强度调整
                adjustment = 1 - alpha * constraint.strength * (1 - is_consistent)
                weights[option] *= adjustment
    
    # 归一化
    total = sum(weights.values())
    if total > 0:
        weights = {k: v/total for k, v in weights.items()}
    
    return weights
```

#### 算法 2: 自适应约束强度

```python
def adaptive_alpha(question_type: str,
                  response_history: List[str],
                  base_alpha: float = 0.8) -> float:
    """
    自适应调整约束强度
    
    Args:
        question_type: 问题类型 (logical/attitudinal/preferential)
        response_history: 历史响应列表
        base_alpha: 基础约束强度
    
    Returns:
        调整后的约束强度
    """
    # 逻辑问题：高约束
    if question_type == "logical":
        return 1.0
    
    # 态度问题：中等约束
    elif question_type == "attitudinal":
        return base_alpha
    
    # 偏好问题：低约束 (保留多样性)
    elif question_type == "preferential":
        return base_alpha * 0.5
    
    # 检测响应多样性
    diversity = compute_diversity(response_history)
    if diversity < 0.3:  # 多样性过低
        return base_alpha * 0.7  # 降低约束
    
    return base_alpha
```

#### 算法 3: 统计 - 个体动态平衡

```python
class StatisticsIndividualBalancer:
    """
    统计 - 个体动态平衡控制器
    """
    def __init__(self, 
                 population_dist: Distribution,
                 residual_personas: List[ResidualPersona],
                 alpha: float = 0.5):
        self.population_dist = population_dist
        self.personas = residual_personas
        self.alpha = alpha  # 平衡参数 (0=纯统计，1=纯个体)
    
    def balanced_generate(self, 
                         question: SurveyQuestion,
                         n_responses: int) -> List[SurveyResponse]:
        """
        平衡生成：同时优化分布匹配和个体差异
        
        Args:
            question: 调查问题
            n_responses: 生成响应数量
        
        Returns:
            响应列表
        """
        responses = []
        
        for _ in range(n_responses):
            # 随机选择模式
            if random.random() < self.alpha:
                # 个体模式：使用 Persona 残差
                persona = random.choice(self.personas)
                response = persona.generate(question)
            else:
                # 统计模式：从群体分布采样
                attributes = self.population_dist.sample()
                response = self.generate_from_stats(attributes, question)
            
            responses.append(response)
        
        return responses
    
    def calibrate_alpha(self, validation_data: pd.DataFrame) -> float:
        """
        校准平衡参数α
        
        Returns:
            最优α值
        """
        best_alpha = 0.5
        best_score = 0
        
        for alpha in np.linspace(0, 1, 11):
            self.alpha = alpha
            responses = self.balanced_generate("test_question", 100)
            
            # 评估分布匹配
            dist_score = self.evaluate_distribution_match(responses)
            
            # 评估个体差异
            div_score = self.evaluate_diversity(responses)
            
            # 综合得分
            score = 0.5 * dist_score + 0.5 * div_score
            
            if score > best_score:
                best_score = score
                best_alpha = alpha
        
        self.alpha = best_alpha
        return best_alpha
```

### 3.5 评估指标

#### 3.5.1 分布匹配指标 (传统)

| 指标 | 公式 | 解释 | 目标 |
|------|------|------|------|
| **KL Divergence** | $D_{KL}(P\|Q) = \sum P(i)\log\frac{P(i)}{Q(i)}$ | 越低越好 | < 1.5 |
| **Distribution Similarity** | $1 - \frac{D_{KL}}{\max(D_{KL})}$ | 越高越好 | > 0.95 |
| **Cohen's Kappa** | $\kappa = \frac{p_o - p_e}{1 - p_e}$ | 越高越好 | > 0.7 |
| **Exact Match** | $\frac{\text{匹配数}}{\text{总数}}$ | 越高越好 | > 0.6 |

---

#### 3.5.2 一致性指标 (核心贡献)

| 指标 | 描述 | 范围 | 目标 |
|------|------|------|------|
| **ACS (Overall)** | 总体态度一致性分数 | [0, 1] | > 0.95 |
| **ACS (Political)** | 政治领域一致性 | [0, 1] | > 0.98 |
| **ACS (Social)** | 社会领域一致性 | [0, 1] | > 0.90 |
| **ACS (Demographic)** | 人口统计一致性 | [0, 1] | > 0.95 |
| **Constraint Violation Rate** | 约束违反率 | [0, 1] | < 0.05 |

**ACS 计算公式**:
$$
ACS(\hat{R}, C) = \frac{1}{|C|} \sum_{c \in C} \frac{1}{|\hat{R}|} \sum_{\hat{r} \in \hat{R}} \mathbb{I}(\hat{r} \text{ satisfies } c)
$$

---

#### 3.5.3 个体差异指标 (创新点 3)

| 指标 | 公式 | 解释 | 目标 |
|------|------|------|------|
| **Variance Retention** | $R_{var} = \frac{Var_{gen}}{Var_{pop}}$ | 方差保留率 | 0.9-1.1 |
| **Diversity Score** | $1 - \text{Avg Similarity}$ | 多样性分数 | > 0.5 |
| **Individual Consistency** | 个体内跨问题一致性 | [0, 1] | > 0.90 |

```python
def generate_persona_pool(population_data: pd.DataFrame, 
                         n_personas: int = 300) -> List[ResidualPersona]:
    """
    生成 Persona 库
    
    步骤:
    1. 从群体分布采样基础属性
    2. 计算个体残差
    3. 生成 ResidualPersona 对象
    4. 验证多样性
    """
    personas = []

            population_data, 
            ind_sample
        )

        personas.append(persona)
    
    # 验证多样性
    diversity = compute_diversity(personas)
    if diversity < 0.5:
        print(f"Warning: Low diversity ({diversity:.3f})")
    
    return personas
```

### 4.3 数据质量检查

#### 4.3.1 分布检查

```python
def check_distribution_quality(processed_data: pd.DataFrame,
                               original_data: pd.DataFrame) -> Dict:
    """
    检查处理后数据的分布质量
    
    指标:
    - KL 散度
    - 卡方检验
    - 边缘分布对比
    """
    results = {}
    
    for col in processed_data.columns:
        if col in original_data.columns:
            # KL 散度
            kl = kl_divergence(
                original_data[col].value_counts(normalize=True),
                processed_data[col].value_counts(normalize=True)
            )
            results[f'{col}_kl'] = kl
            
            # 卡方检验
            chi2, p = chi2_contingency(
                [original_data[col].value_counts(),
                 processed_data[col].value_counts()]
            )
            results[f'{col}_chi2_p'] = p
    
    return results
```

#### 4.3.2 约束验证

```python
def validate_constraints(constraints: List[ConsistencyConstraint],
                        validation_data: pd.DataFrame) -> Dict:
    """
    验证约束在真实数据上的满足率
    
    目标: 确保约束不是人为构造，而是反映真实规律
    """
    results = {}
    
    for constraint in constraints:
        q1, q2 = constraint.question_pair
        
        if q1 in validation_data.columns and q2 in validation_data.columns:
            # 计算真实数据中的满足率
            satisfaction_rate = compute_satisfaction_rate(
                validation_data[q1],
                validation_data[q2],
                constraint
            )
            
            results[constraint.constraint_id] = {
                'satisfaction_rate': satisfaction_rate,
                'expected': constraint.expected_consistency,
                'gap': abs(satisfaction_rate - constraint.expected_consistency)
            }
    
    return results
```
