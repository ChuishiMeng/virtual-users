## 一、修订对照表

| 模块 | 原设计 | 修订后 | 原因 |
|------|--------|--------|------|
| **ACS指标** | ACS = Σ w_c · I_c | **ACS-DH** = Diversity × Calibration | 无法区分真一致vs mode collapse |

| **校准目标** | 无（追求100%一致） | **ACS_target = ACS_human** | ANES数据：Preference=82%, Behavior=78% |

## 二、ACS-DH 指标修订（核心改动）

### 2.1 原 ACS 问题

```python
# 原ACS（有缺陷）
ACS = Σ w_c · I_c(responses)  # 只看约束满足率

# 问题1: baseline ACS=0.917（重复相同回答→mode collapse）
# 问题2: 无法区分"真的态度一致"和"永远选同一个选项"
# 问题3: 没有与真人校准
```

### 2.2 ACS-DH（Diversity-adjusted + Human-calibrated）

```python
class ACS_DH:
    """
    Attitude Consistency Score - Diversity-adjusted & Human-calibrated
    
    解决三个问题：
    1. Mode collapse检测（Diversity调整）
    2. 与真人校准（Calibration误差）
    3. 分层评估（Identity/Preference/Behavior）
    """
    
    def evaluate(self, responses, constraints, human_baseline=None):
        # Step 1: 基础ACS（约束满足率）
        base_acs = self._constraint_satisfaction(responses, constraints)
        
        # Step 2: Diversity调整（惩罚mode collapse）
        diversity = self._response_diversity(responses)
        diversity_penalty = 0 if diversity >= self.human_entropy else \
                           1 - (diversity / self.human_entropy)
        acs_adjusted = base_acs * (1 - diversity_penalty)
        
        # Step 3: Human校准（与真人ACS对比）
        if human_baseline:
            calibration_error = abs(acs_adjusted - human_baseline)
        else:
            calibration_error = None
        
        return ACS_DH_Result(
            base_acs=base_acs,
            diversity=diversity,
            diversity_penalty=diversity_penalty,
            acs_adjusted=acs_adjusted,
            calibration_error=calibration_error,
            layer_scores={
                'identity': self._eval_layer(responses, 'identity'),
                'preference': self._eval_layer(responses, 'preference'),
                'behavior': self._eval_layer(responses, 'behavior'),
            }
        )
    
    def _response_diversity(self, responses):
        """
        测量响应多样性（信息熵）
        
        Key insight: 如果所有回答都一样，entropy=0（mode collapse）
        真人的entropy > 0（有自然波动）
        """
        from scipy.stats import entropy
        # 对每个问题计算响应分布的熵
        entropies = []
        for q_id in set(r.question_id for r in responses):
            q_responses = [r for r in responses if r.question_id == q_id]
            counts = Counter(r.value for r in q_responses)
            probs = np.array(list(counts.values())) / sum(counts.values())
            entropies.append(entropy(probs))
        return np.mean(entropies)
    
    def _eval_layer(self, responses, layer):
        """分层评估"""
        layer_constraints = [c for c in self.constraints if c.layer == layer]
        return self._constraint_satisfaction(responses, layer_constraints)
```

## 三、Persona 表示修订（修复分类变量问题）

### 3.1 原问题

```python
# 原设计（数学不当）
adjusted_value = base_value + residual * variance
# 问题：性别(1=男,2=女) + 0.5 * 0.3 = 1.15 → 无意义
```

### 3.2 修订：双轨表示

```python
class CalibratedPersona:
    """
    校准Persona：分类型用概率偏移，数值型用加性残差
    """
    
    # 属性分类
    NUMERIC_ATTRS = {'age', 'income', 'education_years'}  # 数值型
    CATEGORICAL_ATTRS = {'gender', 'party_id', 'ethnicity', 'occupation'}  # 分类型
    ORDINAL_ATTRS = {'education_level', 'political_ideology', 'income_bracket'}  # 序数型
    
    def __init__(self, id, population_stats, individual_profile):
        self.id = id
        self.population_stats = population_stats  # P(x|demographics)
        self.individual_profile = individual_profile  # 个体固定属性
        self.shift_params = {}  # 校准偏移参数
    
    def get_attribute(self, attr_name):
        """获取个体属性值"""
        if attr_name in self.individual_profile:
            return self.individual_profile[attr_name]  # 固定属性直接返回
        return self._sample_from_population(attr_name)
    
    def get_response_distribution(self, question, context=None):
        """
        获取对某个问题的响应分布（概率）
        
        Key: 输出是概率分布，不是单一值
        """
        base_dist = self.population_stats.get_response_dist(
            question, self.individual_profile
        )
        
        # 应用校准偏移
        if question.id in self.shift_params:
            shift = self.shift_params[question.id]
            calibrated_dist = self._apply_shift(base_dist, shift)
        else:
            calibrated_dist = base_dist
        
        return calibrated_dist
    
    def _apply_shift(self, base_dist, shift):
        """
        对分类型变量：概率偏移（而非加性残差）
        
        base_dist: {'Dem': 0.4, 'Rep': 0.35, 'Ind': 0.25}
        shift: {'Dem': +0.05, 'Rep': -0.03}  # logit空间偏移
        """
        import numpy as np
        
        # 在logit空间做偏移（保证概率合法）
        logits = {k: np.log(v + 1e-10) for k, v in base_dist.items()}
        for k, s in shift.items():
            if k in logits:
                logits[k] += s
        
        # softmax归一化
        max_logit = max(logits.values())
        exp_logits = {k: np.exp(v - max_logit) for k, v in logits.items()}
        total = sum(exp_logits.values())
        return {k: v / total for k, v in exp_logits.items()}
```

## 四、Calibration Controller（替代 ConsistAgent）

### 4.1 核心设计

```python
class CalibrationController:
    """
    校准控制器：让虚拟用户一致性匹配真人水平
    
    与原ConsistAgent的区别：
    - 目标不是100%一致，而是匹配ACS_human
    - 通过α参数分层控制校准强度
    - 包含噪声注入（模拟人类不一致）
    """
    
    def __init__(self, human_baseline, constraints):
        self.human_baseline = human_baseline  # ANES数据
        self.constraints = constraints  # CQCB约束集
        self.alpha = {
            'identity': 0.95,
            'preference': 0.82,  # 校准到ANES数据
            'behavior': 0.79,   # 校准到ANES数据
        }
    
    def calibrate(self, persona, questions):
        """生成校准后的响应"""
        responses = []
        
        for q in questions:
            # 1. 获取Persona的基础响应分布
            base_dist = persona.get_response_distribution(q)
            
            # 2. 检索相关约束
            constraints = self._retrieve_constraints(q, persona, responses)
            
            # 3. 根据层级确定校准强度
            layer = self._determine_layer(q)
            alpha = self.alpha[layer]
            
            # 4. 约束加权 + 噪声注入
            calibrated_dist = self._apply_calibration(
                base_dist, constraints, alpha
            )
            
            # 5. 采样响应
            response = self._sample(calibrated_dist)
            responses.append(response)
        
        return responses
    
    def _apply_calibration(self, base_dist, constraints, alpha):
        """
        校准核心：
        - alpha=1.0 → 完全遵守约束（强制一致）
        - alpha=0.0 → 完全自由（无约束）
        - alpha=ACS_human → 匹配真人一致性水平
        """
        import numpy as np
        
        # 约束倾向分布
        constrained_dist = self._apply_constraints(base_dist, constraints)
        
        # 线性插值：base ← alpha → constrained
        calibrated = {}
        for k in base_dist:
            calibrated[k] = (1 - alpha) * base_dist[k] + alpha * constrained_dist[k]
        
        # 归一化
        total = sum(calibrated.values())
        return {k: v / total for k, v in calibrated.items()}
```

### 4.2 与 ConsistAgent 的对比

| 维度 | ConsistAgent（原） | Calibration Controller（新） |
|------|-------------------|--------------------------|
| **目标** | 最大化ACS | **匹配ACS_human** |
| **α含义** | 约束强度 | **校准强度（到真人水平）** |
| **噪声** | ❌ 无 | **✅ 注入人类级别噪声** |
| **评估** | ACS越高越好 | **Calibration Error < 5%** |
| **mode collapse** | 未防范 | **Diversity-adjusted指标** |
