## 1. 评估指标实现思路

### 1.1 KL散度计算

用于衡量预测分布与真实分布之间的差异。

```python
def kl_divergence(p_true, p_pred):
    """
    计算KL散度: D_KL(P||Q) = Σ P(x) * log(P(x) / Q(x))
    
    参数:
        p_true: 真实分布 (numpy array)
        p_pred: 预测分布 (numpy array)
    
    返回:
        KL散度值 (float)
    """
    # 添加小常数避免log(0)
    epsilon = 1e-10
    p_true = np.array(p_true) + epsilon
    p_pred = np.array(p_pred) + epsilon
    
    # 归一化
    p_true = p_true / p_true.sum()
    p_pred = p_pred / p_pred.sum()
    
    # KL散度公式
    kl = np.sum(p_true * np.log(p_true / p_pred))
    return kl

# 示例: 比较用户回答分布
true_dist = [0.6, 0.3, 0.1]  # 真实用户选择分布
pred_dist = [0.5, 0.4, 0.1]  # 模型预测分布
kl_score = kl_divergence(true_dist, pred_dist)
```

### 1.2 准确率计算

```python
def calculate_accuracy(y_true, y_pred):
    """
    计算分类准确率
    
    参数:
        y_true: 真实标签列表
        y_pred: 预测标签列表
    
    返回:
        准确率 (float, 0-1)
    """
    correct = sum(1 for t, p in zip(y_true, y_pred) if t == p)
    total = len(y_true)
    return correct / total

# 多选题场景: 部分匹配
def partial_accuracy(y_true, y_pred_list):
    """
    计算部分匹配准确率 (预测在候选答案中)
    """
    hits = sum(1 for t, preds in zip(y_true, y_pred_list) if t in preds)
    return hits / len(y_true)
```

### 1.3 Cohen's Kappa 实现

用于衡量两个评估者的一致性 (考虑偶然一致)。

```python
def cohens_kappa(rater1, rater2):
    """
    计算Cohen's Kappa系数
    
    参数:
        rater1: 评估者1的评分列表
        rater2: 评估者2的评分列表
    
    返回:
        Kappa值 (-1到1, 1表示完全一致)
    """
    from sklearn.metrics import confusion_matrix
    
    # 构建混淆矩阵
    labels = sorted(set(rater1 + rater2))
    cm = confusion_matrix(rater1, rater2, labels=labels)
    n = len(rater1)
    
    # 观察一致率 (P_o)
    p_o = np.trace(cm) / n
    
    # 期望一致率 (P_e) - 偶然一致
    row_sums = cm.sum(axis=1) / n  # rater1的边缘分布
    col_sums = cm.sum(axis=0) / n  # rater2的边缘分布
    p_e = np.sum(row_sums * col_sums)
    
    # Kappa公式
    kappa = (p_o - p_e) / (1 - p_e) if (1 - p_e) != 0 else 0
    return kappa

# 使用sklearn内置函数 (推荐)
from sklearn.metrics import cohen_kappa_score
kappa = cohen_kappa_score(human_labels, virtual_user_labels)
```

### 2.1 纯LLM调用流程

```python
class LLMClient:
    """LLM API调用封装"""
    
    def __init__(self, model_name="gpt-4", api_key=None):
        self.model = model_name
        self.client = OpenAI(api_key=api_key)
    
    def call(self, prompt, temperature=0.7, max_tokens=500):
        """
        调用LLM生成回复
        
        参数:
            prompt: 输入提示词
            temperature: 随机性控制 (0=确定性, 1=创造性)
            max_tokens: 最大输出长度
        
        返回:
            生成的文本
        """
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[{"role": "user", "content": prompt}],
            temperature=temperature,
            max_tokens=max_tokens
        )
        return response.choices[0].message.content


def run_llm_baseline(data, prompt_template, llm_client):
    """
    运行LLM baseline评估
    
    参数:
        data: 测试数据集
        prompt_template: 提示词模板
        llm_client: LLM客户端
    
    返回:
        预测结果列表
    """
    predictions = []
    
    for sample in data:
        # 构建提示词
        prompt = prompt_template.format(
            context=sample['context'],
            question=sample['question'],
            options=sample['options']
        )
        
        # 调用LLM
        response = llm_client.call(prompt)
        
        # 解析答案
        pred = parse_answer(response, sample['options'])
        predictions.append(pred)
    
    return predictions
```

def load_dataset(data_path):

### 3.2 循环评估

```python
def run_evaluation(data, methods, llm_client):
    """
    运行完整评估流程
    
    参数:
        data: 测试数据集
        methods: 评估方法字典 {"simple": SIMPLE_PROMPT, "cot": COT_PROMPT}
        llm_client: LLM客户端
    
    返回:
        评估结果字典
    """
    results = {}
    
    for method_name, prompt_template in methods.items():
        print(f"\n{'='*50}")
        print(f"评估方法: {method_name}")
        print('='*50)
        
        predictions = []
        ground_truths = []
        
        # 循环评估每个样本
        for i, sample in enumerate(data):
            print(f"处理样本 {i+1}/{len(data)}", end='\r')
            
            # 获取预测
            prompt = prompt_template.format(**sample)
            response = llm_client.call(prompt)
            pred = parse_answer(response, sample['options'])
            
            predictions.append(pred)
            ground_truths.append(sample['ground_truth'])
        
        # 计算指标
        accuracy = calculate_accuracy(ground_truths, predictions)
        kappa = cohens_kappa(ground_truths, predictions)
        
        # KL散度 (如果有分布信息)
        kl_scores = []
        for sample, pred in zip(data, predictions):
            if 'user_distribution' in sample:
                true_dist = list(sample['user_distribution'].values())
                # 将单一预测转换为分布
                pred_dist = [1.0 if k == pred else 0.0 
                            for k in sample['user_distribution'].keys()]
                kl_scores.append(kl_divergence(true_dist, pred_dist))
        
        avg_kl = np.mean(kl_scores) if kl_scores else None
        
        # 保存结果
        results[method_name] = {
            'predictions': predictions,
            'accuracy': accuracy,
            'kappa': kappa,
            'avg_kl': avg_kl
        }
        
        print(f"\n准确率: {accuracy:.4f}")
        print(f"Cohen's Kappa: {kappa:.4f}")
        if avg_kl:
            print(f"平均KL散度: {avg_kl:.4f}")
    
    return results
```

### 3.3 结果输出

```python
def save_results(results, output_path):
    """保存评估结果"""
    
    # 保存JSON
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(results, f, ensure_ascii=False, indent=2)
    
    # 生成汇总报告
    report = generate_report(results)
    
    report_path = output_path.replace('.json', '_report.md')
    with open(report_path, 'w', encoding='utf-8') as f:
        f.write(report)
    
    print(f"结果已保存至: {output_path}")


def generate_report(results):
    """生成Markdown格式的评估报告"""
    
    report = "# 评估结果报告\n\n"
    report += "| 方法 | 准确率 | Cohen's Kappa | 平均KL散度 |\n"
    report += "|------|--------|---------------|------------|\n"
    
    for method, metrics in results.items():
        kl_str = f"{metrics['avg_kl']:.4f}" if metrics['avg_kl'] else "N/A"
        report += f"| {method} | {metrics['accuracy']:.4f} | {metrics['kappa']:.4f} | {kl_str} |\n"
    
    return report
```

## 注意事项

1. **API限流**: 实际运行时需考虑API调用频率限制，可添加延迟或使用指数退避
2. **成本控制**: 大规模评估时注意token消耗，可先用小数据集测试
3. **结果解析**: 需要robust的答案解析逻辑，处理LLM输出格式不一致的情况
4. **重复实验**: 建议多次运行取平均 (temperature > 0时结果会有随机性)
5. **随机种子**: 设置随机种子确保可重复性
