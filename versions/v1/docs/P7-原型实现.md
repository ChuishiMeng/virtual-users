```python
# app/services/persona_generator.py
from typing import List, Dict
import json
from app.core.llm import LLMClient
from app.models.persona import Persona
from app.utils.prompts import PERSONA_GENERATION_PROMPT

class PersonaGenerator:
    def __init__(self, llm_client: LLMClient):
        self.llm = llm_client
        
    async def generate_persona(
        self, 
        target_population: str,
        demographics: Dict = None
    ) -> Persona:
        """
        生成单个 Persona
        """
        # 构建 Prompt
        prompt = PERSONA_GENERATION_PROMPT.format(
            target_population=target_population,
            demographics=json.dumps(demographics or {}, ensure_ascii=False)
        )
        
        # 调用 LLM
        response = await self.llm.generate(
            prompt=prompt,
            temperature=0.8,
            max_tokens=1000
        )
        
        # 解析 JSON
        persona_dict = self._parse_response(response)
        
        # 验证一致性
        if not self._validate_consistency(persona_dict):
            # 重新生成
            return await self.generate_persona(target_population, demographics)
        
        # 创建 Persona 对象
        persona = Persona(**persona_dict)
        
        return persona
    
    async def generate_persona_pool(
        self,
        target_population: str,
        demographics_distribution: Dict,
        pool_size: int = 100
    ) -> List[Persona]:
        """
        生成 Persona 池
        """
        personas = []
        
        for i in range(pool_size):
            # 从分布中采样人口统计特征
            sampled_demo = self._sample_demographics(demographics_distribution)
            
            # 生成 Persona
            persona = await self.generate_persona(
                target_population=target_population,
                demographics=sampled_demo
            )
            
            personas.append(persona)
        
        # 检查多样性
        self._check_diversity(personas)
        
        return personas
    
    def _parse_response(self, response: str) -> Dict:
        """
        解析 LLM 响应为 JSON
        """
        try:
            # 提取 JSON 部分
            json_start = response.index('{')
            json_end = response.rindex('}') + 1
            json_str = response[json_start:json_end]
            return json.loads(json_str)
        except Exception as e:
            raise ValueError(f"Failed to parse persona: {e}")
    
    def _validate_consistency(self, persona_dict: Dict) -> bool:
        """
        验证 Persona 一致性
        """
        # 检查年龄与收入的一致性
        age = persona_dict['basic_profile']['age_range']
        income = persona_dict['basic_profile']['income_level']
        
        # 简单规则：18-24 岁不太可能高收入
        if age == "18-24" and income in ["高收入", "30K+"]:
            return False
        
        # 检查价值观与决策风格的一致性
        values = persona_dict['psychographics']['values']
        decision_style = persona_dict['psychographics']['decision_style']
        
        if "性价比" in values and decision_style == "冲动型":
            return False
        
        return True
    
    def _sample_demographics(self, distribution: Dict) -> Dict:
        """
        从分布中采样
        """
        import random
        sampled = {}
        
        for key, dist in distribution.items():
            # 加权随机采样
            items = list(dist.keys())
            weights = list(dist.values())
            sampled[key] = random.choices(items, weights=weights)[0]
        
        return sampled
    
    def _check_diversity(self, personas: List[Persona]):
        """
        检查 Persona 池的多样性
        """
        # 统计各维度分布
        age_dist = {}
        for p in personas:
            age = p.basic_profile.age_range
            age_dist[age] = age_dist.get(age, 0) + 1
        
        # 检查是否有过度集中
        max_ratio = max(age_dist.values()) / len(personas)
        if max_ratio > 0.5:
            print(f"Warning: Age distribution skewed, max ratio: {max_ratio}")
```

#### 3.2.1 知识库构建

```python
# app/services/retriever.py
from typing import List, Dict
from chromadb import Client
from chromadb.config import Settings
from app.core.llm import LLMClient

class KnowledgeRetriever:
    def __init__(self, llm_client: LLMClient, persist_dir: str = "./chromadb"):
        self.llm = llm_client
        self.client = Client(Settings(
            chroma_db_impl="duckdb+parquet",
            persist_directory=persist_dir
        ))
        
        # 创建集合
        self.collections = {
            "domain_knowledge": self.client.get_or_create_collection("domain"),
            "persona_knowledge": self.client.get_or_create_collection("persona"),
            "context_knowledge": self.client.get_or_create_collection("context")
        }
    
    async def index_documents(
        self,
        documents: List[Dict],
        collection_name: str = "domain_knowledge"
    ):
        """
        将文档索引到向量数据库
        """
        collection = self.collections[collection_name]
        
        for i, doc in enumerate(documents):
            # 生成 embedding
            embedding = await self.llm.embed(doc['text'])
            
            # 存储
            collection.add(
                ids=[f"{collection_name}_{i}"],
                embeddings=[embedding],
                metadatas=[doc.get('metadata', {})],
                documents=[doc['text']]
            )
    
    async def retrieve(
        self,
        query: str,
        persona_context: str = None,
        collection_name: str = "domain_knowledge",
        top_k: int = 5
    ) -> List[Dict]:
        """
        检索相关知识
        """
        collection = self.collections[collection_name]
        
        # 生成查询 embedding
        query_embedding = await self.llm.embed(query)
        
        # 如果有 Persona 上下文，合并查询
        if persona_context:
            query_embedding = self._merge_embeddings(
                query_embedding,
                await self.llm.embed(persona_context)
            )
        
        # 检索
        results = collection.query(
            query_embeddings=[query_embedding],
            n_results=top_k
        )
        
        # 格式化结果
        retrieved = []
        for i in range(len(results['ids'][0])):
            retrieved.append({
                'id': results['ids'][0][i],
                'text': results['documents'][0][i],
                'metadata': results['metadatas'][0][i],
                'distance': results['distances'][0][i]
            })
        
        return retrieved
    
    def _merge_embeddings(self, emb1: List[float], emb2: List[float], alpha: float = 0.7) -> List[float]:
        """
        合并两个 embedding
        """
        return [alpha * e1 + (1 - alpha) * e2 for e1, e2 in zip(emb1, emb2)]
```

#### 3.2.2 多源检索策略

```python
class MultiSourceRetriever(KnowledgeRetriever):
    async def retrieve_multi_source(
        self,
        query: str,
        persona_context: str,
        top_k_per_source: int = 3
    ) -> Dict[str, List[Dict]]:
        """
        多源检索
        """
        results = {}
        
        # 领域知识检索
        results['domain'] = await self.retrieve(
            query=query,
            persona_context=persona_context,
            collection_name="domain_knowledge",
            top_k=top_k_per_source
        )
        
        # Persona 知识检索
        results['persona'] = await self.retrieve(
            query=persona_context,
            collection_name="persona_knowledge",
            top_k=top_k_per_source
        )
        
        # 上下文知识检索
        results['context'] = await self.retrieve(
            query=query,
            collection_name="context_knowledge",
            top_k=top_k_per_source
        )
        
        # 融合与去重
        results['merged'] = self._merge_and_dedupe(results)
        
        return results
    
    def _merge_and_dedupe(self, results: Dict[str, List[Dict]]) -> List[Dict]:
        """
        合并并去重
        """
        all_docs = []
        seen_texts = set()
        
        for source, docs in results.items():
            for doc in docs:
                # 简单去重（基于文本相似度）
                text_hash = hash(doc['text'][:100])
                if text_hash not in seen_texts:
                    seen_texts.add(text_hash)
                    doc['source'] = source
                    all_docs.append(doc)
        
        # 按距离排序
        all_docs.sort(key=lambda x: x['distance'])
        
        return all_docs[:10]  # 返回 top 10
```

```python
# app/services/responder.py
from typing import Dict, List
from app.core.llm import LLMClient
from app.models.persona import Persona
from app.models.survey import Question, QuestionType
from app.utils.prompts import RESPONSE_GENERATION_PROMPT

class ResponseGenerator:
    def __init__(self, llm_client: LLMClient):
        self.llm = llm_client
    
    async def generate_response(
        self,
        persona: Persona,
        question: Question,
        retrieved_knowledge: List[Dict] = None,
        temperature: float = 0.7
    ) -> Dict:
        """
        生成单个问题的响应
        """
        # 构建上下文
        context = self._build_context(persona, question, retrieved_knowledge)
        
        # 根据问题类型选择生成策略
        if question.question_type == QuestionType.SINGLE_CHOICE:
            return await self._generate_single_choice(context, question, temperature)
        elif question.question_type == QuestionType.MULTIPLE_CHOICE:
            return await self._generate_multiple_choice(context, question, temperature)
        elif question.question_type == QuestionType.LIKERT_SCALE:
            return await self._generate_likert(context, question, temperature)
        elif question.question_type == QuestionType.OPEN_ENDED:
            return await self._generate_open_ended(context, question, temperature)
        else:
            raise ValueError(f"Unsupported question type: {question.question_type}")
    
    async def _generate_single_choice(
        self,
        context: str,
        question: Question,
        temperature: float
    ) -> Dict:
        """
        生成单选题答案
        """
        prompt = f"""{context}

请根据你的 Persona 特征，从以下选项中选择最符合你的一个选项。

【问题】
{question.question_text}

【选项】
{self._format_options(question.options)}

要求：
1. 只返回选项的字母（A/B/C/D/E/F）
2. 不要解释原因

你的选择："""

        response = await self.llm.generate(
            prompt=prompt,
            temperature=temperature,
            max_tokens=10
        )
        
        # 解析答案
        answer = self._parse_choice(response, question.options)
        
        return {
            "question_id": question.question_id,
            "answer": answer,
            "raw_response": response
        }
    
    async def _generate_multiple_choice(
        self,
        context: str,
        question: Question,
        temperature: float
    ) -> Dict:
        """
        生成多选题答案
        """
        max_selections = question.validation.get('max_selections', 3)
        
        prompt = f"""{context}

请根据你的 Persona 特征，从以下选项中选择最符合你的选项（最多选择 {max_selections} 项）。

【问题】
{question.question_text}

【选项】
{self._format_options(question.options)}

要求：
1. 返回选项的字母，用逗号分隔（例如：A, C, E）
2. 不要选择超过 {max_selections} 项
3. 不要解释原因

你的选择："""

        response = await self.llm.generate(
            prompt=prompt,
            temperature=temperature,
            max_tokens=50
        )
        
        # 解析答案
        answers = self._parse_multiple_choices(response, question.options, max_selections)
        
        return {
            "question_id": question.question_id,
            "answer": answers,
            "raw_response": response
        }
    
    async def _generate_likert(
        self,
        context: str,
        question: Question,
        temperature: float
    ) -> Dict:
        """
        生成量表题答案
        """
        prompt = f"""{context}

请根据你的 Persona 特征，对以下陈述进行评分。

【问题】
{question.question_text}

【量表】
1 = 非常不同意
2 = 不同意
3 = 中立
4 = 同意
5 = 非常同意

要求：
1. 只返回一个数字（1-5）
2. 不要解释原因

你的评分："""

        response = await self.llm.generate(
            prompt=prompt,
            temperature=temperature,
            max_tokens=10
        )
        
        # 解析答案
        score = self._parse_likert(response)
        
        return {
            "question_id": question.question_id,
            "answer": score,
            "raw_response": response
        }
    
    async def _generate_open_ended(
        self,
        context: str,
        question: Question,
        temperature: float
    ) -> Dict:
        """
        生成开放题答案
        """
        min_length = question.validation.get('min_length', 20)
        
        prompt = f"""{context}

请根据你的 Persona 特征，回答以下问题。

【问题】
{question.question_text}

要求：
1. 回答要真实、自然，符合你的性格和表达习惯
2. 回答长度至少 {min_length} 字
3. 可以表达不确定，但要说明原因
4. 回答要体现你的价值观和决策风格

你的回答："""

        response = await self.llm.generate(
            prompt=prompt,
            temperature=temperature,
            max_tokens=300
        )
        
        return {
            "question_id": question.question_id,
            "answer": response,
            "raw_response": response
        }
    
    def _build_context(
        self,
        persona: Persona,
        question: Question,
        retrieved_knowledge: List[Dict]
    ) -> str:
        """
        构建上下文
        """
        context_parts = [
            f"你是一个虚拟用户，Persona 如下：\n{persona.json(indent=2, ensure_ascii=False)}"
        ]
        
        if retrieved_knowledge:
            knowledge_text = "\n\n".join([
                f"【相关知识 {i+1}】\n{k['text']}"
                for i, k in enumerate(retrieved_knowledge[:5])
            ])
            context_parts.append(f"\n【相关领域知识】\n{knowledge_text}")
        
        return "\n".join(context_parts)
    
    def _format_options(self, options: List[str]) -> str:
        """
        格式化选项
        """
        letters = 'ABCDEFGHIJ'
        formatted = []
        for i, option in enumerate(options):
            formatted.append(f"{letters[i]}. {option}")
        return "\n".join(formatted)
    
    def _parse_choice(self, response: str, options: List[str]) -> str:
        """
        解析单选题答案
        """
        # 提取字母
        import re
        match = re.search(r'[A-F]', response.upper())
        if match:
            letter = match.group()
            idx = ord(letter) - ord('A')
            if 0 <= idx < len(options):
                return options[idx]
        
        # 如果解析失败，随机选择
        import random
        return random.choice(options)
    
    def _parse_multiple_choices(
        self,
        response: str,
        options: List[str],
        max_selections: int
    ) -> List[str]:
        """
        解析多选题答案
        """
        import re
        # 提取所有字母
        letters = re.findall(r'[A-F]', response.upper())
        
        # 转换为选项
        selected = []
        for letter in letters[:max_selections]:
            idx = ord(letter) - ord('A')
            if 0 <= idx < len(options):
                selected.append(options[idx])
        
        # 如果解析失败，随机选择
        if not selected:
            import random
            selected = random.sample(options, min(max_selections, len(options)))
        
        return selected
    
    def _parse_likert(self, response: str) -> int:
        """
        解析量表题答案
        """
        import re
        match = re.search(r'[1-5]', response)
        if match:
            return int(match.group())
        return 3  # 默认中立
```

```python
    ) -> List['SurveyResponse']:
        """
        批量模拟问卷
        """
        all_responses = []
        
        for persona in personas:
```

#### 3.4.2 分布对比器

```python
class DistributionComparator:
    def compare_distributions(
        self,
        real_responses: List[Dict],
        virtual_responses: List[SurveyResponse],
        question_id: str
    ) -> Dict:
        """
        对比真实与虚拟响应的分布
        """
        import numpy as np
        from scipy import stats
        
        # 提取答案
        real_answers = [r['answers'][question_id] for r in real_responses]
        virtual_answers = [r.responses[question_id]['answer'] for r in virtual_responses]
        
        # 计算分布
        real_dist = self._compute_distribution(real_answers)
        virtual_dist = self._compute_distribution(virtual_answers)
        
        # KL 散度
        kl_div = stats.entropy(real_dist, virtual_dist)
        
        # JS 散度
        js_div = self._jensen_shannon_divergence(real_dist, virtual_dist)
        
        # 卡方检验
        chi2, p_value = stats.chisquare(virtual_dist, f_exp=real_dist)
        
        return {
            "question_id": question_id,
            "real_distribution": real_dist.tolist(),
            "virtual_distribution": virtual_dist.tolist(),
            "kl_divergence": float(kl_div),
            "js_divergence": float(js_div),
            "chi_square": float(chi2),
            "p_value": float(p_value),
            "is_significant": p_value < 0.05
        }
    
    def _compute_distribution(self, answers: List) -> np.ndarray:
        """
        计算分布
        """
        from collections import Counter
        import numpy as np
        
        counter = Counter(answers)
        total = len(answers)
        
        # 归一化
        values = list(counter.values())
        return np.array(values) / total
    
    def _jensen_shannon_divergence(self, p: np.ndarray, q: np.ndarray) -> float:
        """
        计算 JS 散度
        """
        m = 0.5 * (p + q)
        return 0.5 * (stats.entropy(p, m) + stats.entropy(q, m))
```

```python
import asyncio
from typing import List

async def generate_personas_concurrent(
    generator: PersonaGenerator,
    target_population: str,
    demographics_distribution: Dict,
    pool_size: int = 100,
    batch_size: int = 10
) -> List[Persona]:
    """
    并发生成 Persona
    """
    personas = []
    
    for i in range(0, pool_size, batch_size):
        batch_tasks = [
            generator.generate_persona(target_population, demographics_distribution)
            for _ in range(min(batch_size, pool_size - i))
        ]
        
        batch_personas = await asyncio.gather(*batch_tasks)
        personas.extend(batch_personas)
    
    return personas
```
